import java.util.Scanner;
public class Pattern03
{
public static void main(String args[])
{
  int n, i, j, k=65;  
  System.out.println(" Enter the number of the columns:"); 
  Scanner sc=new Scanner(System.in);
  n=sc.nextInt();
  for(i=1;i<=n;i++)
  {
    for(j=1;j<=i;j++)
    {
    System.out.print((char) k + " ");
    k++;;
    }
    System.out.println();
    }
}
}
